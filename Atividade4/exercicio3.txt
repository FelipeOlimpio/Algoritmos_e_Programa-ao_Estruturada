#include <stdio.h>

int potencia(int base, int expoente) {
    int resultado = 1;
    for(int i = 0; i < expoente; i++) {
        resultado *= base;
    }
    return resultado;
}

int main() {
    int base, expoente;
    
    printf("digite a base: ");
    scanf("%d", &base);
    
    printf("digite o expoente (maior ou igual a 1): ");
    scanf("%d", &expoente);
    
    printf("%d elevado a %d Ã©: %d\n", base, expoente, potencia(base, expoente));
    return 0;
}
