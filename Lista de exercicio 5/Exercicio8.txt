#include <stdio.h>

int main() {
    int numeros[10];
    int i, j, temp;

    
    for (i = 0; i < 10; i++) {
        printf("Digite o nÃºmero %d: ", i + 1);
        scanf("%d", &numeros[i]);
    }

    
    for (i = 0; i < 9; i++) {
        for (j = i + 1; j < 10; j++) {
            if (numeros[i] > numeros[j]) {
                // Troca
                temp = numeros[i];
                numeros[i] = numeros[j];
                numeros[j] = temp;
            }
        }
    }

    
    printf("vetor ordenado em ordem crescente:\n");
    for (i = 0; i < 10; i++) {
        printf("%d ", numeros[i]);
    }
    printf("\n");

    return 0;
}
