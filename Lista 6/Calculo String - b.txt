#include <stdio.h>

int calcularTamanho(char str[]) {
    int i = 0;
    while (str[i] != '\0') {
        i++;
    }
    return i;
}

int main() {
    char str[100];
    printf("Digite uma string: ");
    scanf("%99s", str);

    int tamanho = calcularTamanho(str);
    printf("O tamanho da string (sem usar strlen) Ã©: %d\n", tamanho);

    return 0;
}
